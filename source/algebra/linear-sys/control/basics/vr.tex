%
% Системы линейных уравнений
%

% $test: true

\variant{$n+1$}

\begin{problems}

\item
Решите систему:
\[ \left\{\begin{aligned}&
    - 3 x + 5 y - 3 z = 14
\\&
    x + 7 y - 2 z = 7
\\&
    2 x + 7 y + 2 z = 1
\end{aligned}\right.\]

\item
Решите систему; выясните, при каких $a$, $b$ система не имеет решений, а при
каких решений бесконечно много:
\[\left\{\begin{aligned}&
    a x + b y = 1
\\&
    a^2 x + b^2 y = 11
\end{aligned}\right.\]

\end{problems}
