% Контрольная работа: системы линейных уравнений
%

\variant{1}

\begin{problems}

\item
Решите систему:
\[\left\{\begin{aligned}&
    - 2 x - 2 y + 4 z = - 10
\\&
    - 5 x - y - 4 z = 3
\\&
    2 x + 3 y - z = 4
\end{aligned}\right.\]

\item
Решите систему; выясните, при каких $a$, $b$ система не имеет решений, а при
каких решений бесконечно много:
\[\left\{\begin{aligned}&
    a x - b y = a^2
\\&
    a b x + b y = -2
\end{aligned}\right.\]

\end{problems}
