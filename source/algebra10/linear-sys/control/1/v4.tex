% Контрольная работа: системы линейных уравнений
%

\variant{4}

\begin{problems}

\item
Решите систему:
\[\left\{\begin{aligned}&
    - 4 x - 2 y - 4 z = - 4
\\&
    3 x - 5 y + 4 z = 17
\\&
    - 3 x + 3 y + z = - 8
\end{aligned}\right.\]

\item
Решите систему; выясните, при каких $a$, $b$ система не имеет решений, а при
каких решений бесконечно много:
\[\left\{\begin{aligned}&
    b^2 x - a y = -6
\\&
    b x + 3 a y = a^2
\end{aligned}\right.\]

\end{problems}
