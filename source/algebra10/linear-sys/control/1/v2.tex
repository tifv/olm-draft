% Контрольная работа: системы линейных уравнений
%

\variant{2}

\begin{problems}

\item
Решите систему:
\[\left\{\begin{aligned}&
    2 x - y + z = -4
\\&
    4 x - 5 y - 5 z = -4
\\&
    - y - 5 z = 4
\end{aligned}\right.\]

\item
Решите систему; выясните, при каких $a$, $b$ система не имеет решений, а при
каких решений бесконечно много:
\[\left\{\begin{aligned}&
    a x + b y = b^2
\\&
    a x + a b y = 2
\end{aligned}\right.\]

\end{problems}
