% Контрольная работа: системы линейных уравнений
%

\variant{3}

\begin{problems}

\item
Решите систему:
\[\left\{\begin{aligned}&
    - 2 x - 5 y - 4 z = 8
\\&
    - 3 x + y = 1
\\&
    - 4 x - y + 2 z = 8
\end{aligned}\right.\]

\item
Решите систему; выясните, при каких $a$, $b$ система не имеет решений, а при
каких решений бесконечно много:
\[\left\{\begin{aligned}&
    a^2 x + b y = -6
\\&
    a x - 2 b y = b^2
\end{aligned}\right.\]

\end{problems}
